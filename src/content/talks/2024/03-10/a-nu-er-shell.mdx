# Parsing 

```nu 
http get https://raw.githubusercontent.com/elastic/examples/refs/heads/master/Common%20Data%20Formats/apache_logs/apache_logs 
    | save http-logs.txt   

let template = '{ip} - - [{date}] "{method} {url} {protocol}" {status} {size} "{served}" "{useragent}"' 

let logs = open http-logs.txt 
    | lines 
    | parse $template

$logs
    | where method == POST
    | select url status useragent
    | print 

$logs
    | where method == GET
    | where url =~ ".png"
    | where size != "-"
    | into int size
    | where size > 500_000
    | select url status served size
    | print 
```
